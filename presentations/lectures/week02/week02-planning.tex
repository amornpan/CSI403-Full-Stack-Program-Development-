\section{Agile \& Scrum Methodology}

\begin{frame}{What is Agile?}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \begin{block}{Traditional (Waterfall)}
                \begin{itemize}
                    \item Plan everything upfront
                    \item Build all at once
                    \item Test at the end
                    \item Deliver once
                    \item {\color{spumaroon}\textbf{Slow to change}}
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Agile}
                \begin{itemize}
                    \item Plan incrementally
                    \item Build in iterations
                    \item Test continuously
                    \item Deliver frequently
                    \item {\color{codegreen}\textbf{Embrace change}}
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
    
    \vspace{0.5cm}
    
    \begin{successbox}[Agile Manifesto]
        \textbf{Individuals and interactions} over processes and tools\\
        \textbf{Working software} over comprehensive documentation\\
        \textbf{Customer collaboration} over contract negotiation\\
        \textbf{Responding to change} over following a plan
    \end{successbox}
\end{frame}

\section{User Stories \& Requirements}

\begin{frame}{User Story Format}
    \begin{block}{Template}
        \Large
        \textbf{As a} [role],\\
        \textbf{I want to} [action],\\
        \textbf{So that} [benefit].
    \end{block}
    
    \vspace{0.5cm}
    
    \begin{exampleblock}{Example: Loan Application}
        \textbf{As a} borrower,\\
        \textbf{I want to} apply for a new loan,\\
        \textbf{So that} I can get financing for my needs.
    \end{exampleblock}
    
    \vspace{0.3cm}
    
    \begin{infobox}[Tips]
        \begin{itemize}
            \item Focus on \textbf{user value}, not technical details
            \item Keep stories \textbf{small} and \textbf{testable}
            \item Include \textbf{acceptance criteria}
        \end{itemize}
    \end{infobox}
\end{frame}

\begin{frame}{Acceptance Criteria (Given-When-Then)}
    \begin{block}{User Story: User Login}
        \textbf{As a} registered user,\\
        \textbf{I want to} log in to my account,\\
        \textbf{So that} I can access my loan information.
    \end{block}
    
    \vspace{0.3cm}
    
    \begin{exampleblock}{Acceptance Criteria}
        \small
        \textbf{Given} I am on the login page\\
        \textbf{When} I enter valid username and password\\
        \textbf{Then} I should be redirected to dashboard
        
        \vspace{0.2cm}
        
        \textbf{Given} I am on the login page\\
        \textbf{When} I enter invalid credentials\\
        \textbf{Then} I should see an error message
        
        \vspace{0.2cm}
        
        \textbf{Given} I am logged in\\
        \textbf{When} session expires after 24 hours\\
        \textbf{Then} I should be logged out automatically
    \end{exampleblock}
\end{frame}
