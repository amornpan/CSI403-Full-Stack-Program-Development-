\begin{frame}{What is HTML?}
    \weekheader{3}{Jan 21-23}
    
    \begin{block}{HyperText Markup Language}
        HTML is the standard markup language for creating web pages. It describes the \textbf{structure} of a web page.

\begin{frame}{CSS Syntax}

        \begin{center}
            \texttt{selector \{ property: value; \}}
        \end{center}
    \end{block}

.btn-primary {

    padding: 10px 20px;
    border-radius: 5px;
}

    max-width: 500px;

}

\section{Bootstrap 5}
% ============================================

\begin{frame}{What is Bootstrap?}
    \begin{block}{Bootstrap 5}
        The world's most popular CSS framework for building responsive, mobile-first websites.
    \end{block}
    
    \vspace{0.3cm}
    
    \textbf{Key Features:}
    \begin{itemize}
        \item Pre-built components (buttons, cards, forms, navbars)
        \item Responsive grid system (12 columns)
        \item Utility classes for spacing, colors, display
        \item JavaScript plugins (modals, dropdowns, carousels)
        \item No jQuery required (Bootstrap 5+)
    \end{itemize}
    
    \vspace{0.3cm}
    
    \begin{codebox}[Include Bootstrap]
        \begin{lstlisting}[style=html,basicstyle=\ttfamily\tiny]
<link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css" rel="stylesheet">
<script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js"></script>
        \end{lstlisting}
    \end{codebox}
\end{frame}

\begin{frame}{Responsive Breakpoints}
    \begin{center}
        \small
        \begin{tabular}{|l|c|l|}
            \hline
            \textbf{Breakpoint} & \textbf{Class} & \textbf{Dimensions} \\
            \hline
            Extra small & (none) & $<$ 576px \\
            Small & \texttt{sm} & $\geq$ 576px \\
            Medium & \texttt{md} & $\geq$ 768px \\
            Large & \texttt{lg} & $\geq$ 992px \\
            Extra large & \texttt{xl} & $\geq$ 1200px \\
            XXL & \texttt{xxl} & $\geq$ 1400px \\
            \hline
        \end{tabular}
    \end{center}
